Add a New Use Case Scenario

Example Task: Add a new use case scenario where customers can save favorite products.

Solution:

Use Case Name: "Add to Favorites"

Actor: Customer

Main Flow:

  • The customer opens a product page.
  • The customer clicks the "Add to Favorites" button.
  • The system adds the product to the customer's favorite list.
  • The system displays a success message to the customer.

Alternate Flows:

  • If the product is already in the favorites, the system shows a warning message.

public class Customer {
    private List favorites = new ArrayList<>();

    public void addToFavorites(Product product) {
        if (!favorites.contains(product)) {
            favorites.add(product);
            System.out.println("Product added to favorites.");
        } else {
            System.out.println("This product is already in your favorites.");
        }
    }
}

Implement a New Design Pattern

Example Task: Use the Observer Pattern to notify sellers about stock changes.

Solution:


// Subject Interface
interface Subject {
    void attach(Observer o);
    void detach(Observer o);
    void notifyObservers();
}

// Observer Interface
interface Observer {
    void update();
}

// Concrete Subject
class Inventory implements Subject {
    private List observers = new ArrayList<>();
    private List products = new ArrayList<>();

    public void addProduct(Product product) {
        products.add(product);
        notifyObservers();
    }

    public void removeProduct(Product product) {
        products.remove(product);
        notifyObservers();
    }

    @Override
    public void attach(Observer o) {
        observers.add(o);
    }

    @Override
    public void detach(Observer o) {
        observers.remove(o);
    }

    @Override
    public void notifyObservers() {
        for (Observer observer : observers) {
            observer.update();
        }
    }
}

// Concrete Observer
class ProductObserver implements Observer {
    private Product product;

    public ProductObserver(Product product) {
        this.product = product;
    }

    @Override
    public void update() {
        System.out.println("Product " + product.getDetails() + " has been updated in the inventory.");
    }
}

Implement the Adapter Pattern


// Product Interface
interface Product {
    void getDetails();
    void addToCart();
}

// Concrete Products
class Electronics implements Product {
    @Override
    public void getDetails() {
        System.out.println("Electronics: High-end laptop");
    }

    @Override
    public void addToCart() {
        System.out.println("Electronics added to cart.");
    }
}

class Clothing implements Product {
    @Override
    public void getDetails() {
        System.out.println("Clothing: Blue jeans");
    }

    @Override
    public void addToCart() {
        System.out.println("Clothing added to cart.");
    }
}

// Adapter Class
class ProductAdapter {
    private Product product;

    public ProductAdapter(Product product) {
        this.product = product;
    }

    public void getDetails() {
        product.getDetails();
    }

    public void addToCart() {
        product.addToCart();
    }
}
