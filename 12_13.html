Observer Pattern Report

1. Identify the Place for the Behavioural Design Pattern in the System

In the system, the Observer Pattern can be used to notify different product categories or furniture styles about changes in the inventory or availability. This pattern is particularly useful when we want to minimize the need for polling and dynamically change the receivers of notifications.

2. Describe the Problem That Justifies the Need for the Design Pattern

Problem: Different product categories or furniture styles need to be notified about changes in inventory or availability. Without a proper notification mechanism, each product category would need to constantly check for updates, leading to inefficient code and increased complexity.

3. Describe the Pattern, Its Theory, and Its Place in the System

The Observer Pattern defines a one-to-many dependency between objects so that when one object changes state, all its dependents are notified and updated automatically. In this system, the Observer Pattern can be used to notify different product categories about changes in inventory or availability, ensuring that all relevant parties are updated without the need for constant polling.

5. Create the Source Code That Implements the Pattern

Below is an example of Java code that implements the Observer Pattern:


import java.util.ArrayList;
import java.util.List;

// Subject Interface
interface Subject {
    void attach(Observer o);
    void detach(Observer o);
    void notifyObservers();
}

// Observer Interface
interface Observer {
    void update();
}

// Concrete Subject
class Inventory implements Subject {
    private List<Observer> observers = new ArrayList<>();
    private List<Product> products = new ArrayList<>();

    public void addProduct(Product product) {
        products.add(product);
        notifyObservers();
    }

    public void removeProduct(Product product) {
        products.remove(product);
        notifyObservers();
    }

    @Override
    public void attach(Observer o) {
        observers.add(o);
    }

    @Override
    public void detach(Observer o) {
        observers.remove(o);
    }

    @Override
    public void notifyObservers() {
        for (Observer observer : observers) {
            observer.update();
        }
    }
}

// Concrete Observer
class ProductObserver implements Observer {
    private Product product;

    public ProductObserver(Product product) {
        this.product = product;
    }

    @Override
    public void update() {
        System.out.println("Product " + product.getDetails() + " has been updated in the inventory.");
    }
}

// Product Interface
interface Product {
    String getDetails();
    void addToCart();
}

// Concrete Products
class Electronics implements Product {
    @Override
    public String getDetails() {
        return "Electronics: High-end laptop";
    }

    @Override
    public void addToCart() {
        System.out.println("Electronics added to cart.");
    }
}

class Clothing implements Product {
    @Override
    public String getDetails() {
        return "Clothing: Blue jeans";
    }

    @Override
    public void addToCart() {
        System.out.println("Clothing added to cart.");
    }
}

class Books implements Product {
    @Override
    public String getDetails() {
        return "Books: Design Patterns by Erich Gamma";
    }

    @Override
    public void addToCart() {
        System.out.println("Books added to cart.");
    }
}

// Client Code
public class Client {
    public static void main(String[] args) {
        Inventory inventory = new Inventory();
        Product electronics = new Electronics();
        ProductObserver electronicsObserver = new ProductObserver(electronics);

        inventory.attach(electronicsObserver);
        inventory.addProduct(electronics);
    }
}
